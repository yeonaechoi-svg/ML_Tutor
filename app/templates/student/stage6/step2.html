테스트 데이터 예측

predict()란?
학습이 끝난 모델에 새로운 데이터를 입력하면 예측값을 돌려줍니다.
fit()은 학습, predict()는 예측 — 두 역할을 구분하는 것이 중요합니다.
Python 코드 (Jupyter에 붙여넣기) 
{% if algo == 'kmeans' %}# 테스트 데이터 군집 예측
cluster_pred = model.predict(X_test)

# 예측 결과 확인
import pandas as pd
result_df = pd.DataFrame({
    '군집 번호': cluster_pred[:10]
}, index=range(10))
print("테스트 데이터 예측 결과 (앞 10개):")
print(result_df)
print(f"\n전체 {len(cluster_pred)}개 데이터 예측 완료"){% elif algo == 'linear_regression' %}# 테스트 데이터 예측
y_pred = model.predict(X_test)

# 예측 결과 샘플 확인
import pandas as pd
result_df = pd.DataFrame({
    '실제값': y_test.values[:10],
    '예측값': y_pred[:10].round(2),
    '오차': (y_test.values[:10] - y_pred[:10]).round(2)
})
print("테스트 데이터 예측 결과 (앞 10개):")
print(result_df){% else %}# 테스트 데이터 예측
y_pred = model.predict(X_test)

# 예측 결과 샘플 확인
import pandas as pd
result_df = pd.DataFrame({
    '실제값': y_test.values[:10],
    '예측값': y_pred[:10],
    '정답 여부': ['O' if a == b else 'X' for a, b in zip(y_test.values[:10], y_pred[:10])]
})
print("테스트 데이터 예측 결과 (앞 10개):")
